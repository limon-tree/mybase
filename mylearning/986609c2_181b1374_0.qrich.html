后台获取数据:

/usr/lib/lua/luci/controller/admin/status.lua

function action_load()

luci.http.prepare_content("application/json")


local bwc = io.popen("luci-bwc -l 2>/dev/null")

if bwc then

luci.http.write("[")


while true do

local ln = bwc:read("*l")

if not ln then break end

luci.http.write(ln)

end


luci.http.write("]")

bwc:close()

end

end
